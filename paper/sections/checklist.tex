\section*{NeurIPS Paper Checklist}

\begin{enumerate}

\item {\bf Claims}
    \begin{enumerate}
        \item Do the main claims made in the abstract and introduction accurately reflect the paper's contributions and scope?
        \answerYes{All accuracy claims in Tables~\ref{tab:main-gsm8k}--\ref{tab:cross} are supported by paired bootstrap confidence intervals. Utility claims use corrected end-to-end token costs (Section~\ref{sec:exp:setup}). The ablation study (Table~\ref{tab:ablation}) uses second-pass utility for relative comparison, as noted in the caption. Claims about generality are scoped to the validated benchmarks and models (Section~\ref{sec:conclusion}).}
    \end{enumerate}

\item {\bf Limitations}
    \begin{enumerate}
        \item Does the paper discuss the limitations of the work performed by the authors?
        \answerYes{Section~5 discusses limitations including the restriction to mathematical/logical tasks, the requirement of an initial low-budget pass, and the lack of iterative refinement.}
    \end{enumerate}

\item {\bf Theory}
    \begin{enumerate}
        \item Does the paper include theoretical results?
        \answerNo{This is primarily an empirical contribution. The utility formulation in \eqref{eq:utility} provides a formal objective but we do not prove theoretical guarantees.}
    \end{enumerate}

\item {\bf Experiments}
    \begin{enumerate}
        \item Does the paper include experiments?
        \answerYes{Section~4 reports comprehensive experiments across 3 benchmarks and 2 model scales.}
        \item Does the paper report error bars or confidence intervals?
        \answerYes{All accuracy and utility comparisons include 95\% paired bootstrap confidence intervals (10,000 resamples) in Table~\ref{tab:cross}. All CIs exclude zero.}
        \item Does the paper report the number of parameters, FLOPs, or other computational cost?
        \answerYes{Table~\ref{tab:latency} reports wall-clock latency and throughput. Token counts are reported in all tables.}
        \item Does the paper report the computational cost of the experiments?
        \answerYes{All experiments were conducted on a single NVIDIA A100-80GB GPU. Appendix~A reports full hardware and software details.}
    \end{enumerate}

\item {\bf Reproducibility}
    \begin{enumerate}
        \item Does the paper include the code, data, and instructions needed to reproduce the main experimental results?
        \answerYes{Appendix~A provides complete seed protocols, parser details, and training hyperparameters. The codebase and result artifacts will be released.}
        \item Does the paper report all the main experimental details needed to understand the results?
        \answerYes{Budget tiers, $\lambda$ values, bootstrap parameters, and leave-one-subset-out protocols are fully specified.}
    \end{enumerate}

\item {\bf Open access to data and code}
    \begin{enumerate}
        \item Does the paper promise to release code and/or data?
        \answerYes{We will release the complete codebase, per-sample result CSVs, and controller training scripts.}
    \end{enumerate}

\item {\bf Experimental Setting/Details}
    \begin{enumerate}
        \item Does the paper specify the computing infrastructure used for running the experiments?
        \answerYes{NVIDIA A100-80GB, Python 3.10, PyTorch 2.5.1 (CUDA 12.4), HuggingFace Transformers 4.46.0.}
        \item Does the paper specify all the training and test details necessary to reproduce the results?
        \answerYes{Appendix~A provides seed protocols, replication counts, parser details, and training hyperparameters.}
    \end{enumerate}

\item {\bf Broader Impacts}
    \begin{enumerate}
        \item Does the paper discuss potential broader impacts of their work?
        \answerYes{Reducing compute waste for LLM reasoning has positive environmental and cost implications. We do not identify specific negative societal impacts beyond those inherent to LLM deployment.}
    \end{enumerate}

\item {\bf Safeguards}
    \begin{enumerate}
        \item Does the paper describe safeguards for responsible use?
        \answerNA{Our work optimizes inference efficiency for standard reasoning benchmarks and does not introduce new capabilities that require specific safeguards.}
    \end{enumerate}

\item {\bf Licenses}
    \begin{enumerate}
        \item Are the creators of assets used in the paper properly credited?
        \answerYes{GSM8K, MATH, and BIG-Bench Hard are cited. Qwen3 models are publicly available under their respective licenses.}
    \end{enumerate}

\item {\bf New Assets}
    \begin{enumerate}
        \item Are new assets introduced in the paper well documented?
        \answerYes{The controller training scripts and per-sample result files will be released with documentation.}
    \end{enumerate}

\item {\bf Human Subjects}
    \begin{enumerate}
        \item Does the paper involve crowdsourcing or human subjects?
        \answerNo{}
    \end{enumerate}

\end{enumerate}
